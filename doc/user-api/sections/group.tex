\section*{Group Management (\textit{group})}
	Used to access group management.

\subsection*{\textit{listGroups}}
	List all existing groups.
	{
		\color{blue}
		\begin{verbatim}
{
  "section": "group",
  "request": "listGroups",
  "accessKey": "mykey"
}
		\end{verbatim}
	}
	{
		\color{OliveGreen}
		\begin{verbatim}
{
  "section": "group",
  "request": "listGroups",
  "response": "OK",
  "groups": [
    {
      "groupId": 1,
      "name": "Default Group"
    }
  ]
}
		\end{verbatim}
	}
\subsection*{\textit{getGroup}}
	Get details about a group and its members.
	{
		\color{blue}
		\begin{verbatim}
{
  "section": "group",
  "request": "getGroup",
  "groupId": 1,
  "accessKey": "mykey"
}
		\end{verbatim}
	}
	{
		\color{OliveGreen}
		\begin{verbatim}
{
  "section": "group",
  "request": "getGroup",
  "response": "OK",
  "groupId": 1,
  "name": "Default Group",
  "users": [
    1,
    2,
    3
  ],
  "agents": [
    2,
    3
  ]
}
		\end{verbatim}
	}
